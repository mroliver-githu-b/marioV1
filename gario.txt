      const ctx2 = audio();
      const now = ctx2.currentTime;
      const osc = ctx2.createOscillator();
      const gain = ctx2.createGain();
      osc.type = type;
      osc.frequency.setValueAtTime(freq, now);
      gain.gain.setValueAtTime(0, now);
      gain.gain.linearRampToValueAtTime(volume, now + 0.01);
      gain.gain.linearRampToValueAtTime(0, now + duration);
      osc.connect(gain);
      gain.connect(ctx2.destination);
      osc.start(now);
      osc.stop(now + duration);

    function updatePlayer() {
      if (gameState !== "playing") return;

      if (pipeCooldown > 0) pipeCooldown--;
      if (fireCooldown > 0) fireCooldown--;
      if (player.invincibleTimer > 0) player.invincibleTimer--;

      const left = keys["arrowleft"] || keys["a"];
      const right = keys["arrowright"] || keys["d"];
      const jump = keys["arrowup"] || keys["w"] || keys[" "];
      const down = keys["arrowdown"] || keys["s"];
      const fire = keys["f"] || keys["x"];

      if (left) {
        player.vx -= player.speed;
        player.facing = -1;
      }
      if (right) {
        player.vx += player.speed;
        player.facing = 1;
      }
      if (fire) spawnFireball();

      if (!left && !right) player.vx *= FRICTION;
      player.vx = Math.max(-7, Math.min(7, player.vx));

      if (jump && player.onGround) {
        player.vy = -(player.power === "small" ? 16 : 17);
        player.onGround = false;
        sfxJump();
      }

      player.vy += GRAVITY;
      player.x += player.vx;

      const solidsX = [...activePlatforms(), ...activePipes(), ...activeMovingPlatforms()];
      solidsX.forEach(obj => {
        if (rectsCollide(player, obj)) {
          if (player.vx > 0) player.x = obj.x - player.w;
          else if (player.vx < 0) player.x = obj.x + obj.w;
          player.vx = 0;
        }
      });

      player.y += player.vy;
      player.onGround = false;

      const solidsY = [...activePlatforms(), ...activePipes(), ...activeMovingPlatforms()];
      solidsY.forEach(obj => {
        if (rectsCollide(player, obj)) {
          if (player.vy > 0) {
            player.y = obj.y - player.h;
            player.vy = 0;
            player.onGround = true;
            if (obj.vx) player.x += obj.vx;
          } else if (player.vy < 0) {
            player.y = obj.y + obj.h;
            player.vy = 0;
            if (obj.type === "brick" || obj.type === "question") hitBlockFromBelow(obj);
          }
        }
      });

      if (down && player.onGround) {
        activePipes().forEach(pipe => {
          if (!pipe.enter) return;
          const standing = player.x + player.w > pipe.x &&
                           player.x < pipe.x + pipe.w &&
                           Math.abs((player.y + player.h) - pipe.y) < 8;
          if (standing) enterPipe(pipe);
        });
      }

      if (player.x < 0) player.x = 0;

      if (currentArea === "overworld" && player.y > canvas.height + 300) {
        loseLife();
      }

      if (currentArea === "underground" && player.y > 900) {
        player.x = undergroundSpawn.x;
        player.y = undergroundSpawn.y;
        player.vx = 0;
        player.vy = 0;
      }

      activeCoins().forEach(coin => {
        if (!coin.taken && circleRectCollide(coin, player)) {
          coin.taken = true;
          coinsCollected++;
          score += 100;
          sfxCoin();
          updateUI();
        }
      });

      mushrooms.forEach(m => {
        if (!m.active || m.area !== currentArea) return;
        if (rectsCollide(player, m)) {
          m.active = false;
          if (m.kind === "mushroom") player.power = "big";
          if (m.kind === "flower") player.power = "fire";
          setPlayerSize();
          score += 500;
          sfxPower();
          updateUI();
        }
      });

      activeEnemies().forEach(enemy => {
        if (!enemy.alive) return;
        if (rectsCollide(player, enemy)) {
          const playerBottom = player.y + player.h;
          if (player.vy > 0 && playerBottom - enemy.y < 22) {
            enemy.alive = false;
            player.vy = -10;
            score += 250;
            updateUI();
          } else {
            takeHit();
          }
        }
      });

      if (currentArea === "overworld" && !checkpoint.active && rectsCollide(player, checkpoint)) {
        checkpoint.active = true;
        checkpointActivated = true;
        score += 500;
        updateUI();
        statusText.textContent = "Checkpoint reached!";
        sfxPower();
      }

      if (currentArea === "overworld" && boss.alive && player.x > boss.awakenX) {
        if (rectsCollide(player, boss)) {
          const playerBottom = player.y + player.h;
          if (player.vy > 0 && playerBottom - boss.y < 28) {
            boss.health--;
            player.vy = -12;
            score += 400;
            if (boss.health <= 0) {
              boss.alive = false;
              score += 3000;
              statusText.textContent = "Boss defeated! Reach the flag!";
              sfxWin();
            }
            updateUI();
          } else {
            takeHit();
          }
        }
      }

      bossFireballs.forEach(f => {
        if (!f.active || f.area !== currentArea) return;
        if (rectsCollide(player, f)) {
          f.active = false;
          takeHit();
        }
      });

      if (currentArea === "overworld" && !boss.alive && player.x + player.w > 7480 - 20) {
        gameState = "win";
        score += 1000;
        updateUI();
        statusText.textContent = "You Win! Press R";
        sfxWin();
      }

      const areaWidth = areas[currentArea].width;
      cameraX = Math.max(0, Math.min(player.x - canvas.width * 0.35, areaWidth - canvas.width));
    }

    function updateEnemies() {
      if (gameState !== "playing") return;
      activeEnemies().forEach(enemy => {
        if (!enemy.alive) return;
        enemy.x += enemy.vx;

        const areaLeft = enemy.x - 60;
        const areaRight = enemy.x + enemy.w + 60;
        if (currentArea === "overworld") {
          const footX1 = areaLeft + 8;
          const footX2 = areaRight - 8;
          if (activePits().some(p => footX1 < p.x + p.w && footX2 > p.x)) enemy.vx *= -1;
        }

        [...activePlatforms(), ...activePipes(), ...activeMovingPlatforms()].forEach(obj => {
          if (rectsCollide(enemy, obj)) {
            enemy.vx *= -1;
            if (enemy.vx > 0) enemy.x = obj.x + obj.w + 1;
            else enemy.x = obj.x - enemy.w - 1;
          }
        });
      });
    }

    function updateMovingPlatforms() {
      activeMovingPlatforms().forEach(p => {
        p.x += p.vx;
        if (p.x < p.minX || p.x > p.maxX) p.vx *= -1;
      });
    }

    function updatePowerups() {
      mushrooms.forEach(m => {
        if (!m.active || m.area !== currentArea) return;
        if (m.kind === "flower") return;

        m.vy += 0.6;
        m.x += m.vx;

        [...activePlatforms(), ...activePipes(), ...activeMovingPlatforms()].forEach(obj => {
          if (rectsCollide(m, obj)) {
            if (m.vy > 0) {
              m.y = obj.y - m.h;
              m.vy = 0;
            } else if (m.vx > 0) {
              m.x = obj.x - m.w;
              m.vx *= -1;
            } else if (m.vx < 0) {
              m.x = obj.x + obj.w;
              m.vx *= -1;
            }
          }
        });

        m.y += m.vy;
        [...activePlatforms(), ...activePipes(), ...activeMovingPlatforms()].forEach(obj => {
          if (rectsCollide(m, obj) && m.vy > 0) {
            m.y = obj.y - m.h;
            m.vy = 0;
          }
        });
      });
    }

    function updateFireballs() {
      fireballs.forEach(f => {
        if (!f.active || f.area !== currentArea) return;
        f.vy += 0.3;
        f.x += f.vx;
        f.y += f.vy;

        [...activePlatforms(), ...activePipes(), ...activeMovingPlatforms()].forEach(obj => {
          if (rectsCollide(f, obj)) {
            if (f.vy > 0 && f.y + f.h < obj.y + 20) {
              f.y = obj.y - f.h;
              f.vy = -5;
            } else {
              f.active = false;
            }
          }
        });

        activeEnemies().forEach(enemy => {
          if (enemy.alive && rectsCollide(f, enemy)) {
            enemy.alive = false;
            f.active = false;
            score += 250;
            updateUI();
          }
        });

        if (currentArea === "overworld" && boss.alive && rectsCollide(f, boss)) {
          boss.health--;
          f.active = false;
          score += 300;
          if (boss.health <= 0) {
            boss.alive = false;
            score += 3000;
            statusText.textContent = "Boss defeated! Reach the flag!";
            sfxWin();
          }
          updateUI();
        }

        if (f.x < 0 || f.x > WORLD_WIDTH || f.y > 900) f.active = false;
      });

      bossFireballs.forEach(f => {
        if (!f.active || f.area !== currentArea) return;
        f.x += f.vx;
        if (f.x < 0 || f.x > WORLD_WIDTH) f.active = false;
      });
    }

    function updateBoss() {
      if (currentArea !== "overworld" || !boss.alive || gameState !== "playing") return;
      if (player.x < boss.awakenX) return;

      boss.x += boss.vx;
      boss.jumpTimer++;
      boss.shootTimer++;

      if (boss.x < 7000 || boss.x > 7360) boss.vx *= -1;

      if (boss.jumpTimer > 100) {
        boss.y -= 80;
        boss.jumpTimer = 0;
      }
      if (boss.y < 470) {
        boss.y += 4;
        if (boss.y > 470) boss.y = 470;
      }

      if (boss.shootTimer > 70) {
        boss.shootTimer = 0;
        bossFireballs.push({
          x: boss.x + (boss.vx < 0 ? 0 : boss.w),
          y: boss.y + 30,
          w: 18,
          h: 18,
          vx: boss.vx < 0 ? -4 : 4,
          vy: 0,
          active: true,
          area: "overworld"
        });
        sfxFire();
      }
    }

    function drawPlatforms() {

    function updateUI() {
      scoreText.textContent = "Score: " + score;
      coinText.textContent = "Coins: " + coinsCollected;
      powerText.textContent = "Power: " + (player.power === "small" ? "Small" : player.power === "big" ? "Big Green" : "Fire Green");
      fireText.textContent = "Fireballs: " + (player.power === "fire" ? "Ready" : "No");
      livesText.textContent = "Lives: " + lives;

      if (gameState === "title") statusText.textContent = "Press Enter to Start";
      else if (gameState === "playing") {
        if (!checkpointActivated) statusText.textContent = "Reach the checkpoint and beat the boss!";
      }
    }

    function startGame() {
      worldTheme = worlds[selectedWorld];
      lives = 3;
      score = 0;
      coinsCollected = 0;
      hardReset(false);
      gameState = "playing";
      updateUI();
      statusText.textContent = "Good luck!";
    }

    function hardReset(goTitle = true) {
      player.x = overworldStart.x;
      player.y = overworldStart.y;
      player.vx = 0;
      player.vy = 0;
      player.onGround = false;
      player.facing = 1;
      player.power = "small";
      player.invincibleTimer = 0;
      setPlayerSize();

      if (goTitle) {
        gameState = "title";
        lives = 3;
        score = 0;
        coinsCollected = 0;
      }

      cameraX = 0;
      currentArea = "overworld";
      pipeCooldown = 0;
      fireCooldown = 0;
      checkpointActivated = false;
      checkpoint.active = false;

      boss.x = 7220;
      boss.y = 470;
      boss.vx = -1.2;
      boss.health = 6;
      boss.alive = true;
      boss.jumpTimer = 0;
      boss.shootTimer = 0;

      mushrooms.length = 0;
      fireballs.length = 0;
      bossFireballs.length = 0;

      coins.forEach(c => c.taken = false);
      enemies.forEach((e, i) => {
        e.alive = true;
        const starts = [900, 1580, 2470, 3550, 4520, 6250, 830];
        const dirs = [-1.2, 1.2, -1.4, 1.1, -1.2, -1.3, 1.1];
        e.x = starts[i];
        e.vx = dirs[i];
      });

      platforms.forEach(p => {
        if (p.type === "brick") p.broken = false;
        if (p.type === "question") p.used = false;
      });

      movingPlatforms[0].x = 5200;
      movingPlatforms[1].x = 5480;
      movingPlatforms[2].x = 1250;

      updateUI();
    }

    function loop() {
      if (gameState === "playing") {
        updateMovingPlatforms();
        updatePowerups();
        updateFireballs();
        updateEnemies();
        updateBoss();
        updatePlayer();
      }

      if (gameState === "title") drawTitleScreen();
      else drawGame();

      requestAnimationFrame(loop);
    }

    hardReset(true);
    loop();
